SELECT * FROM EMPLOYEE
ORDER BY salary DESC
LIMIT 1;
2) 
SELECT DEPARTMENT.NAME, EMPLOYEE.DEPARTMENT_ID, EMPLOYEE.SALARY
FROM EMPLOYEE
JOIN DEPARTMENT ON EMPLOYEE.DEPARTMENT_ID = DEPARTMENT.ID
ORDER BY salary DESC
LIMIT 1;
3)
SELECT DEPARTMENT.NAME, EMPLOYEE.DEPARTMENT_ID, SUM(EMPLOYEE.SALARY) AS SUM_SALARY
FROM EMPLOYEE
JOIN DEPARTMENT ON EMPLOYEE.DEPARTMENT_ID = DEPARTMENT.ID
GROUP BY DEPARTMENT.NAME
ORDER BY SUM_SALARY DESC
LIMIT 1;
4)
SELECT *
FROM EMPLOYEE
WHERE name LIKE "Р%" AND name LIKE "%н";
